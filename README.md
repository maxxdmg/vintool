cubic_yardage = (square_ft / 9) / (36 / depth)

loose_yardage = cubic_yardage * fluff_value
(NOTE: fluff_value can be 1.5, 1.8. 1.3, or 2 depending on if the material is 'Asphalt', 'Concrete', 'Dirt', or 'Sand' respectively)
